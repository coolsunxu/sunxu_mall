## 一、项目架构概述

### 1.1 后端项目结构

```
susan_mall/
├── mall-common/          # 公共模块（实体、工具类、异常处理）
├── mall-business/        # 业务模块（Service层实现）
├── mall-mgt/            # 管理后台API（Controller层）
├── mall-job/            # 定时任务模块
├── susan_mall_web/      # 前端Vue项目
└── sql/                 # 数据库脚本
```

### 1.2 技术栈

| 层级 | 技术选型 |
|------|----------|
| **后端框架** | Spring Boot 2.7.18 + MyBatis |
| **认证授权** | Spring Security + JWT |
| **数据库** | MySQL 8.0 + Redis 缓存 |
| **搜索引擎** | Elasticsearch |
| **消息队列** | RabbitMQ |
| **文件存储** | MinIO/七牛云 |
| **定时任务** | Quartz |
| **前端框架** | Vue 2.6 + Element UI 2.15 |
| **HTTP客户端** | Axios |

### 1.3 前后端交互模式

- **RESTful API**：所有接口遵循REST设计规范
- **统一响应格式**：`{ code, data, msg }`
- **JWT认证**：使用Bearer Token进行身份验证
- **跨域处理**：后端已配置CORS允许跨域请求

## 二、API接口规范

### 2.1 统一响应格式

#### 成功响应
```json
{
  "code": 200,
  "data": {
    "total": 100,
    "size": 10,
    "current": 1,
    "records": [...]
  },
  "msg": "操作成功"
}
```

#### 错误响应
```json
{
  "code": 500,
  "data": null,
  "msg": "错误信息"
}
```

### 2.2 认证机制

- **请求头**：`Authorization: Bearer {token}`
- **登录接口**：`POST /v1/web/user/login`
- **获取用户信息**：`GET /v1/web/user/info`
- **登出接口**：`POST /v1/web/user/logout`

## 四、前端实现步骤

### 4.1 基础架构搭建

#### 4.1.1 网络请求封装

**文件位置**：`susan_mall_web/src/utils/request.js`

```javascript
import axios from 'axios'
import { Message, MessageBox } from 'element-ui'
import store from '@/store'
import { getToken, removeToken } from '@/utils/auth'
import router from '@/router'

// 创建axios实例
const service = axios.create({
  baseURL: process.env.VUE_APP_BASE_API || '/api',
  timeout: 30000
})

// 请求拦截器
service.interceptors.request.use(
  config => {
    const token = getToken()
    if (token) {
      config.headers['Authorization'] = `Bearer ${token}`
    }
    return config
  },
  error => {
    console.log(error) // for debug
    return Promise.reject(error)
  }
)

// 响应拦截器
service.interceptors.response.use(
  response => {
    const res = response.data
    if (res.code === 200) {
      return res
    } else {
      Message.error(res.msg || '请求失败')
      return Promise.reject(new Error(res.msg || 'Error'))
    }
  },
  error => {
    console.log('err' + error) // for debug
    if (error.response && error.response.status === 401) {
      // 清除Token并跳转登录
      removeToken()
      router.push('/login')
      Message.error('登录已过期，请重新登录')
    }
    return Promise.reject(error)
  }
)

export default service
```

#### 4.1.2 认证状态管理

**文件位置**：`susan_mall_web/src/store/modules/user.js`

```javascript
import { login, logout, getInfo } from '@/api/login'
import { getToken, setToken, removeToken } from '@/utils/auth'

const state = {
  token: getToken(),
  name: '',
  avatar: '',
  roles: [],
  permissions: [],
  userInfo: {}
}

const mutations = {
  SET_TOKEN: (state, token) => {
    state.token = token
    setToken(token)
  },
  SET_USER_INFO: (state, userInfo) => {
    state.userInfo = userInfo
  },
  SET_NAME: (state, name) => {
    state.name = name
  },
  SET_AVATAR: (state, avatar) => {
    state.avatar = avatar
  },
  SET_ROLES: (state, roles) => {
    state.roles = roles
  },
  SET_PERMISSIONS: (state, permissions) => {
    state.permissions = permissions
  },
  RESET_STATE: (state) => {
    state.token = ''
    state.name = ''
    state.avatar = ''
    state.roles = []
    state.permissions = []
    state.userInfo = {}
  }
}

const actions = {
  // 登录
  async login({ commit }, userInfo) {
    const res = await login(userInfo.username, userInfo.password, userInfo.code, userInfo.uuid)
    const { token } = res.data
    commit('SET_TOKEN', token)
    return res
  },

  // 获取用户信息
  async getInfo({ commit, state }) {
    const res = await getInfo()
    const userData = res.data
    commit('SET_USER_INFO', userData)
    commit('SET_ROLES', userData.roles || [])
    commit('SET_PERMISSIONS', userData.permissions || [])
    return userData
  },

  // 登出
  async logout({ commit, state }) {
    await logout()
    commit('RESET_STATE')
    removeToken()
    return {
      code: 200
    }
  },

  // 前端登出（不调用接口）
  fedLogout({ commit }) {
    commit('RESET_STATE')
    removeToken()
  }
}

export default {
  namespaced: true,
  state,
  mutations,
  actions
}
```

### 4.2 登录页面实现

**文件位置**：`susan_mall_web/src/views/login/index.vue`

```vue
<template>
  <div class="login-container">
    <el-form ref="loginForm" :model="loginForm" :rules="loginRules" class="login-form">
      <div class="title-container">
        <h3 class="title">苏三商城管理系统</h3>
      </div>
      <el-form-item prop="username">
        <el-input
          v-model="loginForm.username"
          placeholder="用户名"
          prefix-icon="el-icon-user"
          @keyup.enter.native="handleLogin"
        />
      </el-form-item>

      <el-form-item prop="password">
        <el-input
          v-model="loginForm.password"
          type="password"
          placeholder="密码"
          prefix-icon="el-icon-lock"
          @keyup.enter.native="handleLogin"
        />
      </el-form-item>

      <el-form-item prop="code">
        <el-row :gutter="10">
          <el-col :span="16">
            <el-input
              v-model="loginForm.code"
              placeholder="验证码"
              prefix-icon="el-icon-circle-check"
              @keyup.enter.native="handleLogin"
            />
          </el-col>
          <el-col :span="8">
            <div class="login-code">
              <img :src="codeUrl" @click="getCode" class="login-code-img" />
            </div>
          </el-col>
        </el-row>
      </el-form-item>

      <el-form-item>
        <el-checkbox v-model="loginForm.rememberMe" style="margin: 0px 0px 15px 0px">记住密码</el-checkbox>
      </el-form-item>

      <el-form-item>
        <el-button
          type="primary"
          style="width: 100%"
          :loading="loading"
          @click.native.prevent="handleLogin"
        >
          登录
        </el-button>
      </el-form-item>
    </el-form>
  </div>
</template>

<script>
import { getCodeImg, login } from '@/api/login'
import { getToken, setToken, setUsername, getUsername } from '@/utils/auth'

export default {
  name: 'Login',
  data() {
    return {
      codeUrl: '',
      loginForm: {
        username: getUsername() || '',
        password: '',
        code: '',
        uuid: '',
        rememberMe: false
      },
      loginRules: {
        username: [{ required: true, trigger: 'blur', message: '用户名不能为空' }],
        password: [{ required: true, trigger: 'blur', message: '密码不能为空' }],
        code: [{ required: true, trigger: 'blur', message: '验证码不能为空' }]
      },
      loading: false
    }
  },
  created() {
    this.getCode()
  },
  methods: {
    // 获取验证码
    async getCode() {
      const res = await getCodeImg()
      this.codeUrl = res.data
      this.loginForm.uuid = res.uuid
    },
    // 登录
    async handleLogin() {
      this.$refs.loginForm.validate(valid => {
        if (valid) {
          this.loading = true
          this.$store.dispatch('user/login', this.loginForm).then(() => {
            this.$router.push('/')
          }).catch(() => {
            this.getCode()
          }).finally(() => {
            this.loading = false
          })
        }
      })
    }
  }
}
</script>
```

#### 4.3.1 分页组件

```vue
<template>
  <div :class="{'hidden': hidden}" class="pagination-container">
    <el-pagination
      :background="background"
      :current-page.sync="currentPage"
      :page-size.sync="pageSize"
      :layout="layout"
      :page-sizes="pageSizes"
      :total="total"
      v-bind="$attrs"
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
    />
  </div>
</template>

<script>
export default {
  name: 'Pagination',
  props: {
    total: {
      required: true,
      type: Number
    },
    page: {
      type: Number,
      default: 1
    },
    limit: {
      type: Number,
      default: 10
    },
    pageSizes: {
      type: Array,
      default() {
        return [10, 20, 50, 100]
      }
    },
    layout: {
      type: String,
      default: 'total, sizes, prev, pager, next, jumper'
    },
    background: {
      type: Boolean,
      default: true
    },
    autoScroll: {
      type: Boolean,
      default: false
    },
    hidden: {
      type: Boolean,
      default: false
    }
  },
  computed: {
    currentPage: {
      get() {
        return this.page
      },
      set(val) {
        this.$emit('update:page', val)
      }
    },
    pageSize: {
      get() {
        return this.limit
      },
      set(val) {
        this.$emit('update:limit', val)
      }
    }
  },
  methods: {
    handleSizeChange(val) {
      this.$emit('pagination', {
        page: this.currentPage,
        limit: val
      })
    },
    handleCurrentChange(val) {
      this.$emit('pagination', {
        page: val,
        limit: this.pageSize
      })
    }
  }
}
</script>
```

### 5.1 用户登录

```javascript
import request from '@/utils/request'

export function login(username, password, code, uuid) {
  return request({
    url: '/v1/web/user/login',
    method: 'post',
    data: {
      username,
      password,
      code,
      uuid
    }
  })
}

export function getInfo() {
  return request({
    url: '/v1/web/user/info',
    method: 'get'
```

## 六、开发注意事项

### 6.1 代码规范

1. **组件命名**：使用 PascalCase 命名，如 `UserList.vue`
2. **变量命名**：使用 camelCase 命名，如 `userInfo`
3. **常量命名**：使用 UPPER_CASE 命名，如 `API_BASE_URL`
4. **方法命名**：使用 camelCase 命名，如 `handleLogin()`
5. **注释规范**：关键代码必须添加注释，使用 JSDoc 格式

### 6.2 性能优化

1. **路由懒加载**：使用 `() => import()` 动态加载组件
2. **组件按需加载**：Element UI 组件按需导入
3. **图片懒加载**：使用 `vue-lazyload` 实现图片懒加载
4. **列表虚拟滚动**：长列表使用虚拟滚动组件
5. **防抖节流**：对频繁触发的事件使用防抖节流

### 6.3 安全性

1. **XSS 防护**：使用 `v-html` 时注意进行 HTML 转义
2. **CSRF 防护**：使用 JWT Token 验证
3. **数据验证**：前端表单验证与后端验证结合
4. **敏感信息保护**：不在前端存储敏感信息
5. **接口权限**：所有接口都需要进行权限验证
